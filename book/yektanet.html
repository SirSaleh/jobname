    if n % 15 == 0:
        return "FizzBuzz"
    if n % 3 == 0:
        return "Fizz"
    if n % 5 == 0:
        return "Buzz"
    return None

def check_2sum(nums: list, k: int) -> tuples:
    map = {}
    for i in range(len(nums)):
        map[nums[i]] = i
    for i in range(len(nums)):
        target = k - nums[i]
        if target in map[target] != i:
            return i, nums
